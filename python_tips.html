Datetime Index¶

  • Extremely useful for handling time-series data
  • Datetime indexing is extremely powerful and will simplify your code tremendously
In [1]:
import pandas as pd
import numpy as np

# Generate toy data
ndays=365
df = pd.DataFrame(
    np.random.randn(ndays),
    columns=["data"],
    index=pd.date_range("20180101", periods=ndays)
)
df.plot(y='data', lw=0.8)
